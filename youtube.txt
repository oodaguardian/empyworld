// ══════════════════════════════════════════════
//  FETCH VIDEOS
// ══════════════════════════════════════════════
async function fetchVideos(ch, query = null) {
  const params = new URLSearchParams({
    part: 'snippet',
    maxResults: 20,
    key: API_KEY,
    ...SAFE_PARAMS,
  });

  if (query) {
    // Search mode
    params.set('q', query + ' kids safe');
    params.delete('videoCategoryId');
  } else if (ch.isSearch) {
    params.set('q', ch.query);
    params.delete('videoCategoryId');
  } else {
    // Channel mode
    params.set('channelId', ch.channelId);
    params.delete('videoCategoryId');
    params.delete('type'); // search endpoint uses type
  }

  const endpoint = 'search';
  const url = `${BASE}/${endpoint}?${params}`;

  const res = await fetch(url);
  const data = await res.json();
  if (data.error) throw new Error(data.error.message);
  return data.items || [];
}

// ══════════════════════════════════════════════
//  RENDER

function showSkeletons(container, count = 6) {
  let html = '<div class="loading-grid">';
  for (let i = 0; i < count; i++) {
    html += `
      <div class="skeleton-card">
        <div class="skeleton-thumb"></div>
        <div class="skeleton-line"></div>
        <div class="skeleton-line short"></div>
      </div>`;
  }
  html += '</div>';
  container.innerHTML = html;
}

function renderVideos(items, container, title, pill) {
  if (!items.length) {
    container.innerHTML = `
      <div class="empty-state">
        <div class="emoji">🔍</div>
        <p>No videos found — try another search!</p>
      </div>`;
    return;
  }

  let html = `
    <div class="section-header">
      <div class="section-title">${title}</div>
      <div class="section-pill">${pill}</div>
    </div>
    <div class="video-grid">`;

  items.forEach(item => {
    const vid   = item.id?.videoId || item.id;
    const snip  = item.snippet;
    const thumb = snip?.thumbnails?.medium?.url || snip?.thumbnails?.default?.url || '';
    const titleText = snip?.title || 'Video';
    const channel   = snip?.channelTitle || '';
    const pubDate   = snip?.publishedAt ? new Date(snip.publishedAt).toLocaleDateString('en-US', { month: 'short', year: 'numeric' }) : '';

    html += `
      <div class="video-card" onclick="openPlayer('${vid}', ${JSON.stringify(titleText).replace(/'/g, "\\'")} )">
        <div class="thumb-wrap">
          <img src="${thumb}" alt="${titleText}" loading="lazy" />
          <div class="play-overlay"><div class="play-btn">▶</div></div>
        </div>
        <div class="video-info">
          <div class="video-title">${titleText}</div>
          <div class="video-meta">
            <span>${channel}</span>
            <span class="dot">•</span>
            <span>${pubDate}</span>
          </div>
        </div>
      </div>`;
  });

  html += '</div>';
  container.innerHTML = html;
}

// ══════════════════════════════════════════════
//  LOAD CONTENT
// ══════════════════════════════════════════════
async function loadContent(searchQuery = null) {
  const main = document.getElementById('mainContent');
  showSkeletons(main);

  const ch = activeChannel;
  const label = searchQuery ? `Results for "${searchQuery}"` : ch.label;
  const pill  = searchQuery ? '🔍 Search' : ch.emoji + ' Channel';

  try {
    const items = await fetchVideos(ch, searchQuery);
    renderVideos(items, main, label, pill);
  } catch (err) {
    main.innerHTML = `
      <div class="empty-state">
        <div class="emoji">⚠️</div>
        <p>Oops! ${err.message}</p>
        <p style="margin-top:8px;font-size:0.8rem;opacity:0.7">Check your API key or try again.</p>
      </div>`;
  }
}

function closePlayer() {
  document.getElementById('playerModal').classList.remove('open');
  document.getElementById('playerFrame').src = '';
}

document.getElementById('modalClose').addEventListener('click', closePlayer);
document.getElementById('playerModal').addEventListener('click', e => {
  if (e.target === e.currentTarget) closePlayer();
});

// ══════════════════════════════════════════════
//  SEARCH
// ══════════════════════════════════════════════
function doSearch() {
  const q = document.getElementById('searchInput').value.trim();
  if (!q) return;
  searchMode = true;
  loadContent(q);
}

document.getElementById('searchBtn').addEventListener('click', doSearch);
document.getElementById('searchInput').addEventListener('keydown', e => {
  if (e.key === 'Enter') doSearch();
});

// ══════════════════════════════════════════════
//  VOICE RECOGNITION
// ══════════════════════════════════════════════
const voiceBtn    = document.getElementById('voiceBtn');
const voiceStatus = document.getElementById('voiceStatus');
const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;

if (SpeechRecognition) {
  const recog = new SpeechRecognition();
  recog.continuous    = false;
  recog.interimResults = true;
  recog.lang          = 'en-US';

  recog.onstart = () => {
    voiceBtn.classList.add('listening');
    voiceStatus.textContent = '🎤 Listening... say what you want to watch!';
  };

  recog.onresult = e => {
    let interim = '', final = '';
    for (let i = e.resultIndex; i < e.results.length; i++) {
      const t = e.results[i][0].transcript;
      e.results[i].isFinal ? (final += t) : (interim += t);
    }
    document.getElementById('searchInput').value = final || interim;
    if (final) {
      voiceStatus.textContent = `✅ Got it: "${final}"`;
      setTimeout(() => { voiceStatus.textContent = ''; doSearch(); }, 600);
    }
  };

  recog.onerror = e => {
    voiceBtn.classList.remove('listening');
    voiceStatus.textContent = e.error === 'not-allowed'
      ? '🚫 Microphone blocked — please allow access'
      : '⚠️ Voice error: ' + e.error;
    setTimeout(() => voiceStatus.textContent = '', 3000);
  };

  recog.onend = () => {
    voiceBtn.classList.remove('listening');
    if (!voiceStatus.textContent.startsWith('✅'))
      voiceStatus.textContent = '';
  };

  voiceBtn.addEventListener('click', () => {
    voiceBtn.classList.contains('listening') ? recog.stop() : recog.start();
  });
} else {
  voiceBtn.title = 'Voice search not supported in this browser';
  voiceBtn.style.opacity = '0.4';
  voiceBtn.addEventListener('click', () => {
    voiceStatus.textContent = '⚠️ Voice search requires Chrome or Edge';
    setTimeout(() => voiceStatus.textContent = '', 3000);
  });
}
